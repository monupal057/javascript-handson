Q2- Write code to explain prototype chaining

                const grandParent = {
                    SayGp : function(){
                        console.log("GrandParent Say GP to everyone");
                    }
                }
                
                const Dad = Object.create(grandParent);
                Dad.SayP = function(){
                    console.log("Parent Say Hello To Everyone");
                }
                
                const son = Object.create(Dad);
                son.SayMan = function(){
                    console.log("Say Hi Man to Everyone");
                }
                
                son.SayGp();
                son.SayP();
                son.SayMan();

                Output:
                GrandParent Say GP to everyone
                Parent Say Hello To Everyone
                Say Hi Man to Everyone

Q4- Write a JavaScript function to retrieve all the names of object's own and inherited properties.

                function getAllProperties(obj){
                    let props = [];
                
                    props = Object.getOwnPropertyNames(obj);
                
                    let proto = Object.getPrototypeOf(obj);
                    while(proto){
                        props = props.concat(Object.getOwnPropertyNames(proto));
                        proto = Object.getPrototypeOf(proto);
                    }
                    return props;
                }
